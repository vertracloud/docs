---
title: 'Application Details'
sidebarTitle: 'Get Application'
api: 'GET https://api.vertracloud.app/v1/apps/{id}'
authMethod: 'bearer'
description: 'Retrieves detailed information about an application by its ID.'
---

<ParamField path="id" type="string" required>
  The ID of the application to retrieve details for. If the application belongs to an organization, include the organization ID in the format `appId-orgId`.
</ParamField>

### Response

<ResponseField name="status" type="string">
  Indicates whether the call was successful. `success` if successful, `error` if not.
</ResponseField>

<ResponseField name="response" type="object">
  The details of the application.

  <Expandable title="Toggle object">

  <ResponseField name="id" type="string">
    The application’s ID.
  </ResponseField>

  <ResponseField name="cluster" type="integer">
    The cluster ID of the application.
  </ResponseField>

  <ResponseField name="type" type="integer">
    The application’s type.
  </ResponseField>

  <ResponseField name="name" type="string">
    The application’s name.
  </ResponseField>

  <ResponseField name="description" type="string">
    The application’s description.
  </ResponseField>

  <ResponseField name="owner_id" type="string">
    The ID of the application’s owner.
  </ResponseField>

  <ResponseField name="owner_plan_id" type="integer">
    The ID of the owner’s plan.
  </ResponseField>

  <ResponseField name="language" type="string">
    The application’s programming language (e.g., `nodejs`).
  </ResponseField>

  <ResponseField name="ram" type="integer">
    The application’s RAM usage in MB.
  </ResponseField>

  <ResponseField name="status" type="string">
    The application’s status (e.g., `up`).
  </ResponseField>

  <ResponseField name="subdomain" type="string | null">
    The application’s subdomain, if applicable.
  </ResponseField>

  <ResponseField name="custom_domain" type="string | null">
    The application’s custom domain, if applicable.
  </ResponseField>

  <ResponseField name="last_commit" type="string" format="date-time">
    The timestamp of the application’s last commit.
  </ResponseField>

  <ResponseField name="last_backup" type="string | null" format="date-time">
    The timestamp of the application’s last backup, if applicable.
  </ResponseField>

  <ResponseField name="created_at" type="string" format="date-time">
    The timestamp when the application was created.
  </ResponseField>

  <ResponseField name="updated_at" type="string" format="date-time">
    The timestamp when the application was last updated.
  </ResponseField>

  </Expandable>
</ResponseField>

### Error Responses

<ResponseField name="status" type="string">
  Indicates whether the call was successful. `error` if not successful.
</ResponseField>

<ResponseField name="code" type="string">
  The error code indicating the reason for failure. Possible values:
  - `UNAUTHORIZED`: The user is not authenticated or lacks sufficient permissions.
  - `APP_NOT_FOUND`: The specified application does not exist.
</ResponseField>

<ResponseExample>

```json Error Response (404)
{
  "status": "error",
  "code "APP_NOT_FOUND"
}
```

```json Error Response (403)
{
  "status": "error",
  "code": "UNAUTHORIZED"
}
```

</ResponseExample>
