---
title: Get Application Snapshots
sidebarTitle: Application Snapshots
api: GET https://api.vertracloud.app/v1/apps/:id/commits
authMethod: bearer
description: Retrieves a list of commits for a specified application.
---

{/* @see APIResourceSnapshot[] — @vertracloud/api-types */}
<ParamField path="id" type="string">
  The ID of the application to retrieve commits for.
</ParamField>

### Response

<ResponseField name="response" type="array">
  An array of commit objects.

  <Expandable title="Toggle object">

  <ResponseField name="id" type="string">
    The commit ID.
  </ResponseField>

  <ResponseField name="author_id" type="string">
    The ID of the user who authored the commit.
  </ResponseField>

  <ResponseField name="message" type="string">
    The commit message.
  </ResponseField>

  <ResponseField name="version" type="string">
    The version associated with the commit.
  </ResponseField>

  <ResponseField name="status" type="string">
    The status of the commit.
  </ResponseField>

  <ResponseField name="size" type="string">
    The size of the commit in MB.
  </ResponseField>

  <ResponseField name="date" type="string">
    The timestamp when the commit was created.
  </ResponseField>

  </Expandable>
</ResponseField>

<ResponseExample>

```json Response
{
  "response": [
    {
      "id": "commit123",
      "author_id": "user123",
      "message": "Initial commit",
      "version": "1.0.0",
      "size": "1.23 MB",
      "date": "2026-03-14T21:00:00.000Z"
    }
  ]
}
```

</ResponseExample>

### Error Responses

<ResponseField name="code" type="string">
  The error code indicating the reason for failure. Possible values:
  - `UNAUTHORIZED`: The user is not authenticated or lacks sufficient permissions.
  - `FORBIDDEN`: The user does not have access to the application.
  - `APP_NOT_FOUND`: The specified application does not exist.
</ResponseField>

<ResponseField name="message" type="string">
  A descriptive message providing additional details about the error.
</ResponseField>

<ResponseExample>

```json Error Response (401)
{
  "code": "UNAUTHORIZED"
}
```

```json Error Response (403)
{
  "code": "FORBIDDEN"
}
```

```json Error Response (404)
{
  "code": "APP_NOT_FOUND"
}
```

</ResponseExample>
