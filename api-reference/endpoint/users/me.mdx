---
title: User Information
sidebarTitle: User Info
api: GET https://api.vertracloud.app/v1/users/@me
authMethod: bearer
description: Returns complete user information including applications, databases, plan, OAuth connections, and credits.
---

{/* @see APIUserInfoResponse — @vertracloud/api-types */}

### Response

<ResponseField name="response" type="APIUserInfoResponse">
  Complete authenticated user data.

  <Expandable title="Toggle object">

  <ResponseField name="id" type="string">
    Unique user ID (Snowflake).
  </ResponseField>

  <ResponseField name="email" type="string">
    User email.
  </ResponseField>

  <ResponseField name="name" type="string">
    User display name.
  </ResponseField>

  <ResponseField name="plan_id" type="integer">
    Numeric plan ID. Values: 1=FREE, 2=ECONOMY, 3=PRO, 4=SCALE, 9=INTERMEDIARY, 5=ENTERPRISE_4, 19=ENTERPRISE_6, 6=ENTERPRISE_8, 10=ENTERPRISE_12, 11=ENTERPRISE_14, 7=ENTERPRISE_16, 12=ENTERPRISE_18, 13=ENTERPRISE_20, 14=ENTERPRISE_22, 15=ENTERPRISE_24, 16=ENTERPRISE_26, 17=ENTERPRISE_28, 18=ENTERPRISE_30, 8=ENTERPRISE_32.
  </ResponseField>

  <ResponseField name="language" type="string">
    User's preferred language (`pt-br`, `en-us`, `es-es`).
  </ResponseField>

  <ResponseField name="created_at" type="string" format="date-time">
    Account creation date (ISO 8601).
  </ResponseField>

  <ResponseField name="updated_at" type="string" format="date-time">
    Account last update date (ISO 8601).
  </ResponseField>

  <ResponseField name="plan" type="APIUserPlan">
    Current plan information.

    <Expandable title="Toggle object">

    <ResponseField name="id" type="integer">
      Plan ID.
    </ResponseField>

    <ResponseField name="name" type="string">
      Plan name (e.g., `free`, `pro`, `scale`).
    </ResponseField>

    <ResponseField name="expires_at" type="string | null" format="date-time">
      Plan expiration date. `null` for plans without expiration.
    </ResponseField>

    <ResponseField name="duration" type="integer">
      Plan duration in days.
    </ResponseField>

    <ResponseField name="memory" type="APIUserPlanMemory">
      Plan memory usage.

      <Expandable title="Toggle object">

      <ResponseField name="limit" type="integer">
        Memory limit in MB.
      </ResponseField>

      <ResponseField name="used" type="integer">
        Memory currently in use in MB.
      </ResponseField>

      </Expandable>
    </ResponseField>

    </Expandable>
  </ResponseField>

  <ResponseField name="applications" type="APIApplication[]">
    List of user applications.

    <Expandable title="Toggle object">

    <ResponseField name="id" type="string">
      Unique application ID (Snowflake).
    </ResponseField>

    <ResponseField name="cluster" type="integer">
      ID of the cluster hosting the application.
    </ResponseField>

    <ResponseField name="type" type="integer">
      Application type. `1` = Bot, `2` = Website.
    </ResponseField>

    <ResponseField name="name" type="string">
      Application name.
    </ResponseField>

    <ResponseField name="description" type="string | undefined">
      Application description.
    </ResponseField>

    <ResponseField name="language" type="string">
      Programming language (`javascript`, `typescript`, `python`, `go`, `php`, `ruby`, `java`, `rust`, `bun`, `static`).
    </ResponseField>

    <ResponseField name="ram" type="integer">
      Memory allocated in MB.
    </ResponseField>

    <ResponseField name="status" type="string">
      Current status: `up` or `down`.
    </ResponseField>

    <ResponseField name="subdomain" type="string | null">
      Application subdomain.
    </ResponseField>

    <ResponseField name="custom_domain" type="string | null">
      Custom domain, if configured.
    </ResponseField>

    <ResponseField name="main_file" type="string">
      Application main file.
    </ResponseField>

    <ResponseField name="version" type="string">
      Runtime version (`recommended`, `latest`, or specific version).
    </ResponseField>

    <ResponseField name="auto_restart" type="boolean">
      Whether auto-restart is enabled.
    </ResponseField>

    <ResponseField name="use_credits" type="boolean">
      Whether the application uses credits mode.
    </ResponseField>

    <ResponseField name="last_snapshot" type="string | null" format="date-time">
      Date of the last snapshot.
    </ResponseField>

    <ResponseField name="offline_since" type="string | null" format="date-time">
      Date since the application is offline. `null` if online.
    </ResponseField>

    <ResponseField name="created_at" type="string" format="date-time">
      Creation date.
    </ResponseField>

    <ResponseField name="updated_at" type="string" format="date-time">
      Last update date.
    </ResponseField>

    </Expandable>
  </ResponseField>

  <ResponseField name="databases" type="APIDatabase[]">
    List of user databases.

    <Expandable title="Toggle object">

    <ResponseField name="id" type="string">
      Unique database ID (Snowflake).
    </ResponseField>

    <ResponseField name="cluster" type="integer">
      Cluster ID.
    </ResponseField>

    <ResponseField name="type" type="integer">
      Type: `1` = PostgreSQL, `2` = MongoDB, `3` = Redis, `4` = MySQL.
    </ResponseField>

    <ResponseField name="name" type="string">
      Database name.
    </ResponseField>

    <ResponseField name="description" type="string">
      Database description.
    </ResponseField>

    <ResponseField name="status" type="string">
      Current status: `up` or `down`.
    </ResponseField>

    <ResponseField name="ram" type="integer">
      Memory allocated in MB.
    </ResponseField>

    <ResponseField name="host" type="string">
      Host address for connection.
    </ResponseField>

    <ResponseField name="port" type="integer">
      Connection port.
    </ResponseField>

    <ResponseField name="use_credits" type="boolean">
      Whether the database uses credits mode.
    </ResponseField>

    <ResponseField name="last_snapshot" type="string | null" format="date-time">
      Date of the last snapshot.
    </ResponseField>

    <ResponseField name="offline_since" type="string | null" format="date-time">
      Date since offline.
    </ResponseField>

    <ResponseField name="created_at" type="string" format="date-time">
      Creation date.
    </ResponseField>

    <ResponseField name="updated_at" type="string" format="date-time">
      Last update date.
    </ResponseField>

    </Expandable>
  </ResponseField>

  <ResponseField name="connections" type="APIUserConnection[]">
    Linked OAuth accounts.

    <Expandable title="Toggle object">

    <ResponseField name="provider" type="string">
      OAuth provider (`discord`, `github`, `google`).
    </ResponseField>

    <ResponseField name="username" type="string">
      Username on the provider.
    </ResponseField>

    <ResponseField name="created_at" type="string" format="date-time">
      Binding date.
    </ResponseField>

    </Expandable>
  </ResponseField>

  <ResponseField name="credits" type="APIUserCredits">
    User credits balance, split into compute and AI.

    <Expandable title="Toggle object">

    <ResponseField name="compute" type="APICreditCompute">
      Compute credits (for running apps and databases).

      <Expandable title="Toggle object">

      <ResponseField name="free" type="number">
        Free compute credits balance.
      </ResponseField>

      <ResponseField name="paid" type="number">
        Paid compute credits balance.
      </ResponseField>

      <ResponseField name="free_per_week" type="number">
        Weekly allocation of free credits.
      </ResponseField>

      <ResponseField name="free_reset_at" type="string" format="date-time">
        Date of next free credits reset.
      </ResponseField>

      <ResponseField name="paid_expires_at" type="string | null" format="date-time">
        Paid credits expiration date.
      </ResponseField>

      </Expandable>
    </ResponseField>

    <ResponseField name="ai" type="APICreditAI">
      AI credits (for AI features).

      <Expandable title="Toggle object">

      <ResponseField name="free" type="number">
        Free AI credits balance.
      </ResponseField>

      <ResponseField name="paid" type="number">
        Paid AI credits balance.
      </ResponseField>

      <ResponseField name="free_per_week" type="number">
        Weekly allocation of free AI credits.
      </ResponseField>

      <ResponseField name="free_reset_at" type="string" format="date-time">
        Date of next free AI credits reset.
      </ResponseField>

      <ResponseField name="paid_expires_at" type="string | null" format="date-time">
        Paid AI credits expiration date.
      </ResponseField>

      </Expandable>
    </ResponseField>

    </Expandable>
  </ResponseField>

  </Expandable>
</ResponseField>
